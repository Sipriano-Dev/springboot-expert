
# Criar Network:
docker network create library-network

# Criar container Postgre na versão 16.3
docker run --name postgres -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=library -p 15432:5432 -d --network library-network postgres:18.3


# Criar Pgadmin 4
docker run --name pgadmin4 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=admin -p 25432:80 -d --network library-network dpage/pgadmin4:9.13.0
# Iniciar ambos:
docker start <name>

# Stopa container
docker stop <name>

# Mostra container ativos
docker ps

# Mostra container ativo e não ativo
docker ps -a

# Mostrar todas as portas utilizadas
netstat -aof
# Procurar porta especifica, se não aparecer é que a porta ta livre
netstat -aof | findstr 5432
